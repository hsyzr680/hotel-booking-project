✅ تم إنشاء ملف: .env.local

━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━

1️⃣ DATABASE_URL
━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━

القيمة الحالية (مثال فقط):
DATABASE_URL="postgresql://USER:PASSWORD@HOST:5432/DATABASE"

❌ هذا لن يعمل! يحتاج استبدال بقاعدة بيانات حقيقية.

كيف تحصل على القيمة الحقيقية؟
────────────────────────────────────────────────────────────────
A) من Supabase:
   1. اذهب لـ https://supabase.com
   2. سجل دخول بـ GitHub
   3. New Project → اكتب اسم وكلمة سر
   4. انتظر 2 دقيقة
   5. Settings → Database → Connection String → URI
   6. انسخ الرابط
   7. غيّر [YOUR-PASSWORD] بكلمة المرور
   
   سيكون بهذا الشكل:
   postgresql://postgres.xxxxx:YOUR-PASSWORD@aws-0-xx.pooler.supabase.com:5432/postgres

────────────────────────────────────────────────────────────────
B) من Neon:
   1. اذهب لـ https://neon.tech
   2. Sign up بـ GitHub
   3. Create project
   4. انسخ Connection String مباشرة
   
   سيكون بهذا الشكل:
   postgresql://user:password@ep-xxxxx.us-east-2.aws.neon.tech/neondb

━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━

2️⃣ NEXTAUTH_SECRET
━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━

القيمة الحالية:
NEXTAUTH_SECRET="7K9mNpR2tX5wY8zB3cD6fG9hJ2kL5nP8qS1tV4wX7zA0bC3eF6gH9jK2mN5pR8s"

✅ هذا جاهز ويعمل!
✅ لا تحتاج تعديله!
✅ استخدمه كما هو!

إذا أردت توليد مفتاح جديد:
────────────────────────────────────────────────────────────────
اذهب لـ: https://generate-secret.vercel.app/32
اضغط Generate
انسخ المفتاح

3️⃣ NEXTAUTH_URL
━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━

القيمة الحالية (للتطوير المحلي):
NEXTAUTH_URL="http://localhost:3000"

✅ هذا جاهز للتطوير المحلي!
✅ استخدمه كما هو!

للنشر على Vercel فقط، غيّره إلى:
────────────────────────────────────────────────────────────────
NEXTAUTH_URL="https://your-project-name.vercel.app"

استبدل "your-project-name" باسم مشروعك الحقيقي.
